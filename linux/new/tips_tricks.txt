# List all environment variables
env  

# Run a command with a custom variable
env MY_VAR=123 bash -c 'echo $MY_VAR'  # Output: 123  

# Run a command with a clean environment
env -i sh -c 'echo $PATH'  # Output: (empty)  

# Remove a variable for a command
env -u HOME bash -c 'echo $HOME'  # Output: (empty)
